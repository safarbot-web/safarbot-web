import { Qwen } from 'qwen-sdk'; // Fictional SDK ‚Äî we‚Äôll use direct HTTP

const qwenApiKey = process.env.QWEN_API_KEY;
const model = 'qwen-max';

export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Faqat POST so'rovlar' });
  }

  const { prompt } = req.body;

  // üåç YOUR CUSTOM PROMPT LOGIC FOR UZBEK USERS
  const systemPrompt = `
Siz SafarBot, O'zbekistonliklar uchun sun'iy intellekt asosidagi sayyohlik yordamchisisiz. 
Foydalanuvchi so'roviga asoslanib, batafsil kunlik marshrut tuzing.

Foydalanuvchi so'rovi: "${prompt}"

Aniqlang:
- Chiqish: Toshkent (agar aytilmagan bo'lsa)
- Manzil: masalan, Antalya, Dubay, Bangkok
- Kunlar: raqamini aniqlang
- Oila/bolalar? so'zlar: "oilam", "bolalar" ‚Üí ha
- Byudjet: agar "arzon", "samarali" ‚Üí o'rtacha; "komfort" ‚Üí yuqori

Keyin quyidagilarni taklif qiling:
1. Parvoz yoki avtobus (taxminiy narx, vaqt)
2. Mehmonxona (bolalar uchun mos, O'zbekcha so'zlashuvchi xodimlar bor)
3. Har bir kunga dastur (soatlar bilan)
4. Oziq-ovqat (mahalliy taomlar, O'zbekcha taomlar bor joylar)
5. Viza talablari (O'zbek fuqarolari uchun)
6. Pul almashtirish maslahati

Javobni faqat O'zbek tilida ber, tablitsasiz, odobli uslubda.
  `;

  try {
    const response = await fetch('https://dashscope.aliyuncs.com/api/v1/services/aigc/text-generation/generation', {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${qwenApiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: model,
        input: { messages: [{ role: 'user', content: systemPrompt }] },
        parameters: { temperature: 0.7, max_tokens: 1500 }
      })
    });

    const data = await response.json();
    const aiResponse = data.output?.text || "Kechirasiz, hozircha reja tuza olmayman. Iltimos, qayta urinib ko'ring.";
    
    res.status(200).json({ text: aiResponse });
  } catch (error) {
    console.error(error);
    res.status(500).json({ text: "So'rovni bajarishda xato. Iltimos, keyinroq urinib ko'ring." });
  }
}
